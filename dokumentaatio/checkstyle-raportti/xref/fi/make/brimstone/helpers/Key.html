3   /**
4    * Simple data container class, so that the game can always query, whether a
5    * button has been pressed down, but not released.
6    *
7    * @author make
8    */
9   public class Key {
10  
11      private boolean pressed;
12      private int key;
13      private String name;
14  
15      /**
16       * Constructor initializes the class.
17       * @param name Name of the key.
18       * @param key The keys value.
19       */
20      public Key(String name, int key) {
21          this.name = name;
22          this.key = key;
23          pressed = false;
24      }
25  
26      /**
27       * Key value getter.
28       * @return Key value.
29       */
30      public int getKey() {
31          return key;
32      }
33  
34      /**
35       * Key name getter.
36       * @return Key name.
37       */
38      public String getKeyName() {
39          return name;
40      }
41  
42      /**
43       * Called, when a KeyPressed event for a key of interest is received.
44       */
45      public void press() {
46          pressed = true;
47      }
48  
49      /**
50       * Called when a KeyReleased event for a key of interest is received.
51       */
52      public void release() {
53          pressed = false;
54      }
55  
56      /**
57       * The method called by the game itself, to see if a key is pressed.
58       * @return The status of the key.
59       */
60      public boolean isPressed() {
61          return pressed;
62      }
63  }
