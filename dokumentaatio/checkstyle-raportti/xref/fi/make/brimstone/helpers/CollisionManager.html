8   /**
9    * A helper class for calculating collisions and effects of collisions, in order
10   * to make code in the MapController and Updater classes less convoluted.
11   *
12   * @author make
13   */
14  public class CollisionManager {
15  
16      /**
17       * Simply calculates whether two MapObjects are colliding. NOTE: Assumes
18       * that all MapObjects are 32x32 units large objects.
19       *
20       * @param a Any object that extends the abstract MapObject class.
21       * @param b Any object that extends the abstract MapObject class.
22       * @return True if the two pieces collide.
23       */
24      public static boolean collides(MapObject a, MapObject b) {
25          if (b.getX() - a.getX() < 32
26                  && b.getX() - a.getX() > -32
27                  && b.getY() - a.getY() < 32
28                  && b.getY() - a.getY() > -32) {
29              return true;
30          }
31          return false;
32      }
33  
34      /**
35       * Calculates what direction the player takes after hitting the wall. This
36       * method does not check for collision, so it assumes collision detection
37       * has been made before hand. Direction calculations have been made on the
38       * assumption, that all of the NCU's are 32x32 squares. It simply compares
39       * the dX and dY values of the player and wall and, due to collision being
40       * assumed, the one with the larger absolute value is bound to be the one
41       * causing the collision. Then the player's speed in the direction in
42       * question gets reversed and halved. (Halved in order to stop endless
43       * acceleration in case of close adjacent walls).
44       *
45       * This method also "unsticks" the player from the wall by pushing him back
46       * in the direction he came from.
47       *
48       * @param p Player object.
49       * @param n NCU (Wall) object.
50       */
51      public static void redirectPlayerFromWall(Player p, NCU n) {
52          Vector directionFromWall = new Vector(n.getX() - p.getX(), n.getY() - p.getY());
53  
54          if (Math.abs(directionFromWall.x) == Math.abs(directionFromWall.y)) {    //I wanted to handle the off case that the player approaches directly from a corner
55              p.setSpeed(new Vector(p.getSpeed().x * -0.5, p.getSpeed().y * -0.5));
56          } else {
57              if (Math.abs(directionFromWall.x) > Math.abs(directionFromWall.y)) {
58                  p.setSpeed(new Vector(p.getSpeed().x * -.5, p.getSpeed().y));
59                  if (directionFromWall.x < 0) {
60                      p.setX(n.getX() + 32);
61                  } else {
62                      p.setX(n.getX() - 32);
63                  }
64              } else {
65                  p.setSpeed(new Vector(p.getSpeed().x, p.getSpeed().y * -.5));
66                  if (directionFromWall.y < 0) {
67                      p.setY(n.getY() + 32);
68                  } else {
69                      p.setY(n.getY() - 32);
70                  }
71              }
72          }
73      }
74  
75      /**
76       * Makes sure the player doesn't stick to the level edges due to overlaps.
77       *
78       * @param player Player object.
79       * @param lvl0 Level object.
80       */
81      public static void unStickPlayer(Player player, Level lvl0) {
82          if (player.getX() < 32) {
83              player.setX(32);
84          } else if (player.getX() > lvl0.getLevelDimensions().x - 64) {
85              player.setX(lvl0.getLevelDimensions().x - 64);
86          }
87          if (player.getY() < 32) {
88              player.setY(32);
89          } else if (player.getY() > lvl0.getLevelDimensions().y - 64) {
90              player.setY(lvl0.getLevelDimensions().y - 64);
91          }
92      }
93  }
