1   package fi.make.brimstone.gui;
2   
3   import java.awt.Graphics;
4   import javax.swing.JPanel;
5   import java.awt.Color;
6   import java.awt.Dimension;
7   import java.awt.Toolkit;
8   import java.awt.Font;
9   
10  import fi.make.brimstone.game.MapController;
11  import fi.make.brimstone.game.mapobjects.Player;
12  import fi.make.brimstone.game.mapobjects.MapObject;
13  
14  public class Painter extends JPanel {
15  
16      private Dimension d;
17      private Player plr;
18      private MapController m;
19  
20      public Painter(MapController m, Dimension d) {
21          super.setBackground(Color.GREEN);
22          this.m = m;
23          this.plr = m.getPlayer();
24          this.d = d;
25      }
26  
27      @Override
28      public void paintComponent(Graphics g) {
29          super.paintComponent(g);
30  
31          for (MapObject mo : m.getAllObjects()) {
32              g.drawImage(mo.getImage(), (int) (mo.getX() - plr.getX())
33                      + (d.width / 2), (int) (mo.getY() - plr.getY() + (d.height / 2)), this);
34          }
35  
36          g.setFont(new Font("TimesRoman", Font.PLAIN, 30));
37          g.drawString("Enemies alive: " + Integer.toString(m.getEnemyAmount()), d.width - 300, d.height - 100);
38          g.drawString("Enemies killed: " + Integer.toString(m.getEnemiesKilled()), d.width - 300, d.height - 60);
39  
40          if (m.isPaused()) {
41              g.setFont(new Font("TimesRoman", Font.PLAIN, 60));
42              g.setColor(Color.RED);
43              g.drawString("PAUSED", d.width / 2 - 100, d.height / 2);
44          }
45  
46          Toolkit.getDefaultToolkit().sync();
47      }
48  
49      public void updateWindowSize(Dimension d) {
50          this.d = d;
51      }
52  
53  }
